One-time setup: run this SQL in Supabase → SQL Editor to create the platform invoices table:

CREATE TABLE platform_invoices (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  account_email text,
  account_user_id uuid,
  date date,
  due_date date,
  inv_num text,
  payment text,
  notes text,
  inv_total numeric(10,2) DEFAULT 0,
  inv_bal numeric(10,2) DEFAULT 0,
  service_lines jsonb,
  status_text text,
  failed_charge boolean DEFAULT false,
  deleted boolean DEFAULT false,
  past_due boolean DEFAULT false,
  to_print boolean DEFAULT false,
  to_email boolean DEFAULT false,
  created_at timestamptz DEFAULT now()
);
