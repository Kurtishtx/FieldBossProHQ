Configure automated alerts sent to clients for scheduling events.

One-time setup — run this in Supabase → SQL Editor:

CREATE TABLE IF NOT EXISTS alert_toggles (id uuid DEFAULT uuid_generate_v4() PRIMARY KEY, user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE, alert_type text NOT NULL, sms_enabled boolean DEFAULT false, email_enabled boolean DEFAULT false, UNIQUE(user_id, alert_type)); ALTER TABLE alert_toggles ENABLE ROW LEVEL SECURITY; CREATE POLICY "own" ON alert_toggles FOR ALL USING (auth.uid() = user_id);
