Alert Email Info

This information auto-fills into your alert email templates when sent to clients.

One-time setup — run this SQL in Supabase → SQL Editor:

CREATE TABLE IF NOT EXISTS alert_email_settings (id uuid DEFAULT uuid_generate_v4() PRIMARY KEY, user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE UNIQUE, company_name text, company_logo text, company_address text, company_city text, company_state text, company_zip text, company_phone text, company_email text, company_website text, facebook_url text, instagram_url text, twitter_url text, youtube_url text, show_facebook boolean DEFAULT true, show_instagram boolean DEFAULT true, show_twitter boolean DEFAULT true, show_youtube boolean DEFAULT true, signoff_name text); ALTER TABLE alert_email_settings ENABLE ROW LEVEL SECURITY; CREATE POLICY "own" ON alert_email_settings FOR ALL USING (auth.uid() = user_id);
